# 左移右移

```C++
#include <iostream>
#include <bitset>
#include <iomanip>
#include <ios>

using namespace std;

void print(string desc,int v){
    cout<<desc<<": "<<"(hex)="<<setw(sizeof(int)*2)<<setfill('0')<<hex<<v<<"    ";
    cout<<"(bin)="<<bitset<sizeof(int)*8>(v)<<"    ";
    cout<<"(dec)="<<dec<<v<<endl;
}
int main(){

    cout<<endl<<"--------------------------------------------------------------"<<endl;
    cout<<"1. 原码：最高位表示符号位，正数0，负数1；其余7位表示数值"<<endl;
    cout<<"2. 反码：正数的反码和原码相同，负数的反码为除符号位其余取反"<<endl;
    cout<<"3. 补码：正数的补码和原码相同，负数的补码为反码+1"<<endl;
    cout<<"4. 左移：左边的二进制丢弃，右边补0"<<endl;
    cout<<"5. 右移：保持符号位不变，正数左补0，负数左补1"<<endl;
    cout<<"6. 计算机使用补码来表示数字"<<endl;
    cout<<"--------------------------------------------------------------"<<endl<<endl;

    if(sizeof(int) != 4){
        cout << "the test is request int is 4Bits"<<endl;
        return -1;
    }
    cout<<"current system sizeof(int):"<<sizeof(int)<<"Bits"<<endl;
    cout<<"---------------------------"<<endl;
    int i = 0xFFFFFFFF;
    print("src",i);print("<<1",i<<1);print(">>1",i>>1);
    cout<<"---------------------------"<<endl;

    i = 0x80000000;
    print("src",i);print("<<1",i<<1);print(">>1",i>>1);
    cout<<"---------------------------"<<endl;

    i = 0x7FFFFFFF;
    print("src",i);print("<<1",i<<1);print(">>1",i>>1);
    cout<<"---------------------------"<<endl;

    i = 0x00000000;
    print("src",i);print("<<1",i<<1);print(">>1",i>>1);
    cout<<"---------------------------"<<endl;

    i = 0x00000001;
    print("src",i);print("<<1",i<<1);print(">>1",i>>1);
    cout<<"---------------------------"<<endl;

    return 0;
}
```
